create image: qemu-img create -f raw Image1.img 100G
first boot add: -cdrom ~/code/isos/arch.iso

running:
qemu-system-x86_64 \
-enable-kvm \
-cpu host \
-m 10G \
-mem-prealloc \
-smp 8 \
-drive file=Image1.img,if=none,cache=none,id=drive0,format=raw \
-device virtio-blk-pci,drive=drive0 \
-device virtio-net-pci,netdev=net0 \
-netdev user,id=net0 \
-object iothread,id=iothread0 \
-device virtio-scsi-pci \
